5         body[`offset${axis}`],

7         html[`client${axis}`],
8         html[`offset${axis}`],

12           parseInt(computedStyle[`margin${axis === 'Height' ? 'Top' : 'Left'}`]) + 
13           parseInt(computedStyle[`margin${axis === 'Height' ? 'Bottom' : 'Right'}`]))
14         : 0

16     }
