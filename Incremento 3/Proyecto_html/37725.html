24             //Metodo para convertir una lista en tabla
25             public DataTable ConvertToDataTable<T>(IList<T> data)
26             {
27                 PropertyDescriptorCollection properties =
28                    TypeDescriptor.GetProperties(typeof(T));
29                 DataTable table = new DataTable();
30                 foreach (PropertyDescriptor prop in properties)
31                     table.Columns.Add(prop.Name, Nullable.GetUnderlyingType(prop.PropertyType) ?? prop.PropertyType);
32                 foreach (T item in data)
33                 {
34                     DataRow row = table.NewRow();
35                     foreach (PropertyDescriptor prop in properties)
36                         row[prop.Name] = prop.GetValue(item) ?? DBNull.Value;
37                     table.Rows.Add(row);
38                 }
39                 return table;
40             }
