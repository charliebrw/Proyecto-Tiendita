11     export default function getFixedPositionOffsetParent(element) {
12       // This check is needed to avoid errors in case one of the elements isn't defined for any reason
13        if (!element || !element.parentElement || isIE()) {
14         return document.documentElement;
15       }
16       let el = element.parentElement;
17       while (el && getStyleComputedProperty(el, 'transform') === 'none') {
18         el = el.parentElement;
19       }
20       return el || document.documentElement;
