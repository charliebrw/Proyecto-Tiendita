75             [HttpPost]
76             public ActionResult Edit(Productos producto, HttpPostedFileBase Imagen)
77             {
78                 byte[] ImagenActual = null;
79     
80                 if (Imagen == null)
81                 {
82                     ImagenActual = db.Productos.SingleOrDefault(t => t.IDProducto == producto.IDProducto).proImagen;
83                     db.sp_M_Producto(producto.IDProducto, producto.proNombre, producto.proPrecioVenta, producto.IDCategoria, producto.RUC,ImagenActual);
84                     return RedirectToAction("Index");
85                 }
86                 else
87                 {
88                     producto.proImagen = new byte[Imagen.ContentLength];
89                     Imagen.InputStream.Read(producto.proImagen, 0, Imagen.ContentLength);
90                     db.sp_M_Producto(producto.IDProducto, producto.proNombre, producto.proPrecioVenta, producto.IDCategoria, producto.RUC,producto.proImagen);
91                     return RedirectToAction("Index");
92                 }
